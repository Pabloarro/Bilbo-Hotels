46 public Usuario guardarUsuario(RegistroUsuarioDTO registroDTO) {
47 log.info("Intentando guardar un nuevo usuario: {}", registroDTO.getUsername());
48
49 Optional<Usuario> usuarioExistente = Optional.ofNullable(repoUsuario.findByUsername(registroDTO.getUsername()));
50 if (usuarioExistente.isPresent()) {
51 throw new NombreUsuarioYaExisteException("¡Este nombre de usuario ya está registrado!");
52 }
53
54 Usuario usuario = mapRegistrationDtoToUser(registroDTO);
55
56 if (TipoRol.CLIENTE.equals(registroDTO.getTipoRol())) {
57 Cliente cliente = Cliente.builder().usuario(usuario).build();
58 repoCliente.save(cliente);
59 } else if (TipoRol.HOTEL_MANAGER.equals(registroDTO.getTipoRol())) {
60 HotelManager hotelManager = HotelManager.builder().usuario(usuario).build();
61 repoHotelManager.save(hotelManager);
62 }
63
64 Usuario usuarioGuardado = repoUsuario.save(usuario);
65 log.info("Se ha guardado correctamente el nuevo usuario: {}", registroDTO.getUsername());
66 return usuarioGuardado;
67 }

146 @Override
147 public Usuario resetPassword(ResetearPasswordDTO resetearPasswordDTO) {
148 Authentication auth = SecurityContextHolder.getContext().getAuthentication();
149 String loggedInUsername = auth.getName();
150
151 Usuario usuario = repoUsuario.findByUsername(loggedInUsername);
152 if (usuario == null) {
153 throw new UsernameNotFoundException("Useruario no encontrado");
154 }
155 if (!passwordEncoder.matches(resetearPasswordDTO.getAntiguaPassword(), usuario.getPassword())) {
156 throw new IllegalArgumentException("La antigua contrasenia es incorrecta");
157 }
158 usuario.setPassword(passwordEncoder.encode(resetearPasswordDTO.getNuevaPassword()));
159 return repoUsuario.save(usuario);
160 }
