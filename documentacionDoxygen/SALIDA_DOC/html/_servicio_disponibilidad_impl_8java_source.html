41 @Override
42 public void actualizarDisponibilidades(long hotelId, LocalDate fechaCheckIn, LocalDate fechaCheckOut, List<HabitacionSeleccionDTO> seleccionHabitaciones) {
43 Hotel hotel = servicioHotel.buscarHotelById(hotelId).orElseThrow(() -> new EntityNotFoundException("Hotel no encontrado"));
44 log.info("Intentando actualizar las disponibilidades para el ID del hotel {} desde {} hasta {}", hotelId, fechaCheckIn, fechaCheckOut);
45
46 seleccionHabitaciones = seleccionHabitaciones.stream()
47 .filter(seleccionHabitacion -> seleccionHabitacion.getContar() > 0)
48 .collect(Collectors.toList());
49
50 // Iterate through the room selections made by the user
51 for (HabitacionSeleccionDTO seleccionHabitacion : seleccionHabitaciones) {
52 TipoHabitacion tipoHabitacion = seleccionHabitacion.getTipoHabitacion();
53 int selectedCount = seleccionHabitacion.getContar();
54 log.debug("Procesando {} habitación(es) de tipo {}", selectedCount, tipoHabitacion);
55
56 // Find the room by roomType for the given hotel
57 Habitacion room = hotel.getHabitaciones().stream()
58 .filter(r -> r.getTipoHabitacion() == tipoHabitacion)
59 .findFirst()
60 .orElseThrow(() -> new EntityNotFoundException("Tipo habitacion no encontrada"));
61
62 // Iterate through the dates and update or create availability
63 for (LocalDate fecha = fechaCheckIn; fecha.isBefore(fechaCheckOut); fecha = fecha.plusDays(1)) {
64 final LocalDate fechaActual = fecha; // Temporary final variable
65 Disponibilidad disponibilidad = repoDisponibilidad.findByHabitacionIdAndFecha(room.getId(), fecha)
66 .orElseGet(() -> Disponibilidad.builder()
67 .hotel(hotel)
68 .fecha(fechaActual)
69 .habitacion(room)
70 .habitacionesDisponibles(room.getContadorHabitacion())
71 .build());
72
73 // Reduce the available rooms by the selected count
74 int actualizadasHabitacionesDisponibles = disponibilidad.getHabitacionesDisponibles() - selectedCount;
75 if (actualizadasHabitacionesDisponibles < 0) {
76 throw new IllegalArgumentException("Las habitaciones seleccionadas superan las habitaciones disponibles para la fecha: " + fechaActual);
77 }
78 disponibilidad.setHabitacionesDisponibles(actualizadasHabitacionesDisponibles);
79
80 repoDisponibilidad.save(disponibilidad);
81 }
82 }
83 log.info("Se han actualizado las disponibilidades correctamente para el ID del hotel {} desde {} hasta {}", hotelId, fechaCheckIn, fechaCheckOut);
