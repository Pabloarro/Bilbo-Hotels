108 private void validateBookingDates(LocalDate checkinDate, LocalDate checkoutDate) {
109 if (checkinDate.isBefore(LocalDate.now())) {
110 throw new IllegalArgumentException("La fecha de check in no puede estar en pasado");
111 }
112 if (checkoutDate.isBefore(checkinDate.plusDays(1))) {
113 throw new IllegalArgumentException("La fecha de check out debe ser posterior a la de check in.");
114 }
115 }

117 @Override
118 public ReservaDTO mapReservaModelToReservaDto(Reserva reserva) {
119 DireccionDTO addressDto = DireccionDTO.builder()

125 List<HabitacionSeleccionDTO> roomSelections = reserva.getHabitacionReservadas().stream()

127 .tipoHabitacion(habitacion.getTipoHabitacion())
128 .contar(habitacion.getContar())
129 .build())
130 .collect(Collectors.toList());
131
132 Usuario customerUser = reserva.getCliente().getUsuario();
133
134 return ReservaDTO.builder()
135 .id(reserva.getId())
136 .numeroConfirmacion(reserva.getNumeroConfirmacion())
137 .fechaReserva(reserva.getFechaReserva())

139 .hotelId(reserva.getHotel().getId())
140 .fechaCheckIn(reserva.getFechaCheckIn())
141 .fechaCheckOut(reserva.getFechaCheckOut())
142 .seleccionesHabitacion(roomSelections)
143 .precioTotal(reserva.getPago().getPrecioTotal())
144 .nombreHotel(reserva.getHotel().getNombre())
145 .direccionHotel(addressDto)
146 .nombreCliente(customerUser.getNombre() + " " + customerUser.getApellido())
147 .emailCliente(customerUser.getUsername())
148 .estadoPago(reserva.getPago().getEstadoPago())

153 private Reserva mapBookingInitDtoToBookingModel(InicioReservaDTO inicioReservaDTO, Cliente cliente, Hotel hotel) {
154 Reserva reserva = Reserva.builder()
155 .cliente(cliente)
156 .hotel(hotel)
157 .fechaCheckIn(inicioReservaDTO.getFechaCheckIn())
158 .fechaCheckOut(inicioReservaDTO.getFechaCheckOut())
159 .build();
160
161 for (HabitacionSeleccionDTO roomSelection : inicioReservaDTO.getHabitacionSelecciones()) {
162 if (roomSelection.getContar() > 0) {
163 HabitacionReservada habitacionReservada = HabitacionReservada.builder()
164 .reserva(reserva)
165 .tipoHabitacion(roomSelection.getTipoHabitacion())
166 .contar(roomSelection.getContar())
167 .build();
168 reserva.getHabitacionReservadas().add(habitacionReservada);
169 }
170 }
171
172 return reserva;
173 }
