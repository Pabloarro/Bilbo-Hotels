71 private String registrarUsuario(RegistroUsuarioDTO registroDTO, BindingResult result, String view, String redirectUrl) {
72 if (result.hasErrors()) {
73 log.warn("El registro fallo debido a errores de validacion: {}", result.getAllErrors());
74 return view;
75 }
76 try {
77 servicioUsuario.guardarUsuario(registroDTO);
78 log.info("Registro de usuario exitoso: {}", registroDTO.getUsername());
79 } catch (NombreUsuarioYaExisteException e) {
80 log.error("El registro ha fallado porque el nombre de usuario ya existe: {}", e.getMessage());
81 result.rejectValue("username", "user.exists", e.getMessage());
82 return view;
83 }
84 return "redirect:/" + redirectUrl + "?success";
85 }
