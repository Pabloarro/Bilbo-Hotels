32 @Override
33 @Transactional
34 public Hotel guardarHotel(HotelRegistroDTO hotelRegistroDTO) {
35 log.info("Intentando guardar un nuevo hotel: {}", hotelRegistroDTO.toString());
36
37 Optional<Hotel> hotelExistente = repoHotel.findByNombre(hotelRegistroDTO.getNombre());
38 if (hotelExistente.isPresent()) {
39 throw new HotelYaExisteException("¡Este nombre de hotel ya está registrado!");
40 }
41
42 Hotel hotel = mapHotelRegistrationDtoToHotel(hotelRegistroDTO);
43
44 Direccion direccionGuardada = servicioDireccion.guardarDireccion(hotelRegistroDTO.getDireccionDTO());
45 hotel.setDireccion(direccionGuardada);
46
47 String username = SecurityContextHolder.getContext().getAuthentication().getName();
48 HotelManager hotelManager = servicioHotelManager.findByUsuario(servicioUsuario.buscarUsuarioByUsername(username));
49 hotel.setHotelManager(hotelManager);
50
51 hotel = repoHotel.save(hotel);
52
53 List<Habitacion> habitacionesGuardadas = servicioHabitacion.guardarHabitaciones(hotelRegistroDTO.getHabitacionDTOs(), hotel);
54 hotel.setHabitaciones(habitacionesGuardadas);
55
56 Hotel hotelGuardado = repoHotel.save(hotel);
57 log.info("Se ha guardado correctamente el nuevo hotel con ID: {}", hotel.getId());
58 return hotelGuardado;
59 }

180 @Override
181 public HotelDTO mapHotelToHotelDto(Hotel hotel) {
182 List<HabitacionDTO> habitacionDTOs = hotel.getHabitaciones().stream()
183 .map(servicioHabitacion::mapHabitacionToHabitacionDto) // convert each Room to RoomDTO
184 .collect(Collectors.toList()); // collect results to a list
185
186 DireccionDTO direccionDTO = servicioDireccion.mapDireccionToDireccionDto(hotel.getDireccion());
187
188 return HotelDTO.builder()
189 .id(hotel.getId())
190 .nombre(hotel.getNombre())
191 .direccionDTO(direccionDTO)
192 .habitacionDTOs(habitacionDTOs)
193 .managerUsername(hotel.getHotelManager().getUsuario().getUsername())
194 .build();
195 }
