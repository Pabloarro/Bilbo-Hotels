33 @Override
34 public List<HotelDisponibilidadDTO> buscarHotelesDisponiblesByCiudadAndFecha(String ciudad, LocalDate fechaCheckIn, LocalDate fechaCheckOut) {
35 validateCheckinAndfechaCheckOuts(fechaCheckIn, fechaCheckOut);
36
37 log.info("Intentando encontrar hoteles en {} con habitaciones disponibles desde {} hasta {}", ciudad, fechaCheckIn, fechaCheckOut);
38
39 // Number of days between check-in and check-out
40 Long numeroDias = ChronoUnit.DAYS.between(fechaCheckIn, fechaCheckOut);
41
42 // 1. Fetch hotels that satisfy the criteria (min 1 available room throughout the booking range)
43 List<Hotel> hotelsWithAvailableRooms = repoHotel.findHotelsWithAvailableRooms(ciudad, fechaCheckIn, fechaCheckOut, numeroDias);
44
45 // 2. Fetch hotels that don't have any availability records for the entire booking range
46 List<Hotel> hotelsWithoutAvailabilityRecords = repoHotel.findHotelsWithoutAvailabilityRecords(ciudad, fechaCheckIn, fechaCheckOut);
47
48 // 3. Fetch hotels with partial availability; some days with records meeting the criteria and some days without any records
49 List<Hotel> hotelsWithPartialAvailabilityRecords = repoHotel.findHotelsWithPartialAvailabilityRecords(ciudad, fechaCheckIn, fechaCheckOut, numeroDias);
50
51 // Combine and deduplicate the hotels using a Set
52 Set<Hotel> combinedHotels = new HashSet<>(hotelsWithAvailableRooms);
53 combinedHotels.addAll(hotelsWithoutAvailabilityRecords);
54 combinedHotels.addAll(hotelsWithPartialAvailabilityRecords);
55
56 log.info("Se encontraron {} hoteles con habitaciones disponibles.", combinedHotels.size());
57
58 // Convert the combined hotel list to DTOs for the response
59 return combinedHotels.stream()
60 .map(hotel -> mapHotelToHotelDisponibleDto(hotel, fechaCheckIn, fechaCheckOut))
61 .collect(Collectors.toList());
62 }

81 @Override
82 public HotelDisponibilidadDTO mapHotelToHotelDisponibleDto(Hotel hotel, LocalDate fechaCheckIn, LocalDate fechaCheckOut) {
83 List<HabitacionDTO> habitacionDTOs = hotel.getHabitaciones().stream()
84 .map(servicioHabitacion::mapHabitacionToHabitacionDto) // convert each Room to RoomDTO
85 .collect(Collectors.toList());
86
87 DireccionDTO direccionDTO = servicioDireccion.mapDireccionToDireccionDto(hotel.getDireccion());
88
89 HotelDisponibilidadDTO hotelDisponibilidadDTO = HotelDisponibilidadDTO.builder()
90 .id(hotel.getId())
91 .nombre(hotel.getNombre())
92 .direccionDTO(direccionDTO)
93 .habitacionDTOs(habitacionDTOs)
94 .build();
95
96 // For each room type, find the minimum available rooms across the date range
97 int maxAvailableSingleRooms = hotel.getHabitaciones().stream()
98 .filter(habitacion -> habitacion.getTipoHabitacion() == TipoHabitacion.INDIVIDUAL)
99 .mapToInt(habitacion -> servicioDisponibilidad.getMininimoHabitacionesDisponibles(habitacion.getId(), fechaCheckIn, fechaCheckOut))
100 .max()
101 .orElse(0); // Assume no single rooms if none match the filter
102 hotelDisponibilidadDTO.setMaxHabitacionesIndividualDisponibles(maxAvailableSingleRooms);
103
104 int maxAvailableDoubleRooms = hotel.getHabitaciones().stream()
105 .filter(habitacion -> habitacion.getTipoHabitacion() == TipoHabitacion.DOBLE)
106 .mapToInt(habitacion -> servicioDisponibilidad.getMininimoHabitacionesDisponibles(habitacion.getId(), fechaCheckIn, fechaCheckOut))
107 .max()
108 .orElse(0); // Assume no double rooms if none match the filter
109 hotelDisponibilidadDTO.setMaxHabitacionesDoblesDisponibles(maxAvailableDoubleRooms);
110
111 return hotelDisponibilidadDTO;
112 }
